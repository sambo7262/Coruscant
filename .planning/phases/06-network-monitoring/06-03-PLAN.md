```tsx
function ThroughputBar({ label, value, peak, color }: { label: string; value: number | null; peak: number; color: string }) {
  const pct = value !== null && peak > 0 ? Math.min((value / peak) * 100, 100) : 0
  const display = value !== null ? `${value >= 1 ? Math.round(value) : value.toFixed(1)}M` : '—'
  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '4px' }}>
      <span style={{ fontSize: '8px', color: 'var(--text-offwhite)', fontFamily: 'var(--font-mono)', width: '14px' }}>
        {label}
      </span>
      <div style={{ flex: 1, height: '4px', background: '#222', borderRadius: '2px' }}>
        <div style={{ width: `${pct}%`, height: '100%', background: color, borderRadius: '2px', transition: 'width 0.3s ease' }} />
      </div>
      <span style={{ fontSize: '8px', color: 'var(--text-offwhite)', fontFamily: 'var(--font-mono)', width: '32px', textAlign: 'right' }}>
        {display}
      </span>
    </div>
  )
}
```

**Add UnifiDetailView component** inside ServiceDetailPage.tsx (alongside existing PiholeDetailView, SabnzbdDetailView, ArrDetailView):

```tsx
/** UniFi detail view: devices grouped by type (D-10, D-11, D-12, D-13) */
function UnifiDetailView({ metrics }: { metrics: Record<string, unknown> }) {
  const um = metrics as unknown as UnifiMetrics
  const devices = um?.devices ?? []

  // Classify devices per D-06
  const gateways = devices.filter(d => {
    const m = d.model.toUpperCase()
    return m.startsWith('UDM') || m.startsWith('UDMP') || m.startsWith('UDR')
  })
  const switches = devices.filter(d => d.model.toUpperCase().startsWith('USW'))
  const aps = devices.filter(d => {
    const m = d.model.toUpperCase()
    return m.startsWith('U6') || m.startsWith('UAP') || m.startsWith('UAL') || m.startsWith('UAE')
  })
  // Unrecognized devices go into a catch-all section
  const knownMacs = new Set([...gateways, ...switches, ...aps].map(d => d.macAddress))
  const other = devices.filter(d => !knownMacs.has(d.macAddress))

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '12px', overflowY: 'auto', maxHeight: '100%' }}>
      {/* Summary row */}
      <div style={{ display: 'flex', gap: '16px', fontSize: '10px', fontFamily: 'var(--font-mono)', color: 'var(--text-offwhite)' }}>
        <span>{um?.clientCount ?? 0} clients</span>
        <span>TX {um?.wanTxMbps !== null ? `${Math.round(um.wanTxMbps ?? 0)}M` : '—'}</span>
        <span>RX {um?.wanRxMbps !== null ? `${Math.round(um.wanRxMbps ?? 0)}M` : '—'}</span>
      </div>

      {gateways.length > 0 && <DeviceSection label="GATEWAYS" devices={gateways} />}
      {switches.length > 0 && <DeviceSection label="SWITCHES" devices={switches} />}
      {aps.length > 0 && <DeviceSection label="ACCESS POINTS" devices={aps} />}
      {other.length > 0 && <DeviceSection label="OTHER" devices={other} />}
    </div>
  )
}
```

**Add DeviceRow helper component per D-11:**

```tsx
function DeviceRow({ device }: { device: UnifiDevice }) {
  const isOnline = device.state === 'online'
  // Format uptime as Xd Yh (per D-19)
  const days = Math.floor(device.uptime / 86_400)
  const hours = Math.floor((device.uptime % 86_400) / 3_600)
  const uptimeStr = days > 0 ? `${days}d ${hours}h` : `${hours}h`

  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '6px', padding: '2px 0',
      fontSize: '9px', fontFamily: 'var(--font-mono)', color: 'var(--text-offwhite)' }}>
      {/* LED dot or X for offline per D-11 */}
      {isOnline ? (
        <span style={{ color: 'var(--cockpit-green)', fontSize: '10px' }}>●</span>
      ) : (
        <span style={{ color: 'var(--cockpit-red)', fontSize: '10px' }}>✕</span>
      )}
      {/* Model name */}
      <span>{device.model}</span>
      {/* Uptime — only shown when online */}
      {isOnline && <span style={{ color: '#888' }}>up {uptimeStr}</span>}
      {/* Client count — only shown when online per D-11; right-aligned */}
      {isOnline && device.clientCount > 0 && (
        <span style={{ marginLeft: 'auto' }}>{device.clientCount} clients</span>
      )}
    </div>
  )
}
```
