### Task 2 — Fix disk temp color thresholds

Update `tempColor()` function in AppHeader.tsx:

```ts
// Before (too aggressive — red at 45.6°C which is normal operating temp)
function tempColor(tempF: number): string {
  if (tempF >= 114) return '#FF3B3B'  // red
  if (tempF >= 95)  return '#FF8C00'  // orange
  return 'var(--cockpit-amber)'
}

// After (matches Synology DSM thresholds)
function tempColor(tempF: number): string {
  if (tempF >= 140) return '#FF3B3B'   // red   — ≥ 60°C (DSM critical)
  if (tempF >= 113) return '#FF8C00'   // orange — ≥ 45°C (DSM warning)
  return 'var(--cockpit-amber)'         // amber  — normal
}
```

Same change applies to the CPU temp gauge fill percent — update scale if needed so the bar fills appropriately across the 32°F–158°F (0–70°C) range.
