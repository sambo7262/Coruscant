**Step 1: Add Docker container stats function to NAS adapter (packages/backend/src/adapters/nas.ts)**

Add a new exported function `fetchNasDockerStats` that queries `SYNO.Docker.Container` to get per-container stats and aggregates them:

```typescript
export async function fetchNasDockerStats(
  baseUrl: string,
  username: string,
  password: string,
): Promise<NasDockerStats | undefined> {
  try {
    const sid = await ensureSession(baseUrl, username, password)

    // Try SYNO.Docker.Container first, fall back to SYNO.ContainerManager.Container
    for (const api of ['SYNO.Docker.Container', 'SYNO.ContainerManager.Container']) {
      const params = new URLSearchParams({
        api,
        version: '1',
        method: 'list',
        _sid: sid,
      })

      const response = await axios.get(`${baseUrl}/webapi/entry.cgi?${params.toString()}`, {
        timeout: TIMEOUT_MS,
      })

      if (response.data?.success && response.data?.data?.containers) {
        const containers: Array<{
          cpu_usage?: number        // percentage
          memory_usage?: number     // bytes
          memory_limit?: number     // bytes
          up_bytes?: number         // total network up bytes
          down_bytes?: number       // total network down bytes
          status?: string           // 'running' | 'stopped' etc.
        }> = response.data.data.containers

        const running = containers.filter(c => c.status === 'running')
        if (running.length === 0) return { cpuPercent: 0, ramPercent: 0, networkMbpsUp: 0, networkMbpsDown: 0 }

        const cpuPercent = running.reduce((sum, c) => sum + (c.cpu_usage ?? 0), 0)
        const totalMemUsage = running.reduce((sum, c) => sum + (c.memory_usage ?? 0), 0)
        const totalMemLimit = running.reduce((sum, c) => sum + (c.memory_limit ?? 0), 0)
        const ramPercent = totalMemLimit > 0 ? (totalMemUsage / totalMemLimit) * 100 : 0

        // Network: these are total bytes, not rate — store as-is for now
        // The poll interval can derive rate from delta if needed later
        const networkUp = running.reduce((sum, c) => sum + (c.up_bytes ?? 0), 0)
        const networkDown = running.reduce((sum, c) => sum + (c.down_bytes ?? 0), 0)

        return {
          cpuPercent: Math.round(cpuPercent * 10) / 10,
          ramPercent: Math.round(ramPercent * 10) / 10,
          networkMbpsUp: Math.round((networkUp / 1048576) * 10) / 10,  // MB for display
          networkMbpsDown: Math.round((networkDown / 1048576) * 10) / 10,
        }
      }
    }

    return undefined  // neither API worked
  } catch {
    return undefined
  }
}
```

```typescript
export async function checkNasImageUpdates(baseUrl: string, username: string, password: string): Promise<boolean> {
  try {
    const sid = await ensureSession(baseUrl, username, password)

    // Try both Docker and ContainerManager namespaces
    for (const api of ['SYNO.Docker.Image', 'SYNO.ContainerManager.Image']) {
      for (const version of ['1', '2']) {
        const params = new URLSearchParams({
          api,
          version,
          method: 'list',
          _sid: sid,
        })

        const response = await axios.get(`${baseUrl}/webapi/entry.cgi?${params.toString()}`, {
          timeout: TIMEOUT_MS,
        })

        if (!response.data?.success) continue

        const images: Array<Record<string, unknown>> = response.data?.data?.images ?? []
        // Check multiple possible field names for update availability
        const hasUpdate = images.some((img) =>
          img.is_update_available === true ||
          img.canUpgrade === true ||
          img.upgrade_available === true
        )
        if (hasUpdate) return true
        // If we got a successful response, trust this result
        return false
      }
    }

    return false
  } catch {
    return false
  }
}
```
