**4. Create log-transport.ts (per D-24, D-25):** A `stream.Writable` subclass that receives pino JSON log lines and inserts into `app_logs`. Use the simpler Writable pattern (NOT pino-abstract-transport) to avoid ESM resolution issues:

```typescript
import { Writable } from 'node:stream'
import { getDb } from './db.js'
import { appLogs } from './schema.js'
import { logEvents, type LogEntry } from './log-events.js'

const LEVEL_MAP: Record<number, string> = { 30: 'info', 40: 'warn', 50: 'error' }

export class SqliteLogStream extends Writable {
  _write(chunk: Buffer, _enc: string, cb: () => void) {
    try {
      const line = chunk.toString().trim()
      if (!line) { cb(); return }
      const obj = JSON.parse(line) as Record<string, unknown>
      const numLevel = typeof obj.level === 'number' ? obj.level : 0
      if (numLevel < 30) { cb(); return } // skip debug/trace per D-25
      const levelStr = LEVEL_MAP[numLevel] ?? (numLevel >= 50 ? 'error' : 'info')
      const service = typeof obj.service === 'string' ? obj.service : 'system'
      const message = typeof obj.msg === 'string' ? obj.msg : ''
      const entry: LogEntry = {
        timestamp: new Date().toISOString(),
        level: levelStr,
        service,
        message,
        payload: JSON.stringify(obj),
      }
      const result = getDb().insert(appLogs).values(entry).run()
      entry.id = Number(result.lastInsertRowid)
      logEvents.emit('entry', entry)
    } catch { /* malformed line — skip */ }
    cb()
  }
}
```

```typescript
import { getDb } from '../db.js'
import { kvStore } from '../schema.js'
import { eq } from 'drizzle-orm'

// In-memory cache loaded from kv_store on first access
let cachedPeakTx: number | null = null
let cachedPeakRx: number | null = null
let cachedPeakClients: number | null = null

function getOrLoadPeak(key: string, cached: number | null): number {
  if (cached !== null) return cached
  const row = getDb().select().from(kvStore).where(eq(kvStore.key, key)).get()
  return row ? parseFloat(row.value) : 0
}

function updateHighWaterMark(key: string, current: number, cached: number | null): number {
  const stored = getOrLoadPeak(key, cached)
  if (current > stored) {
    getDb().insert(kvStore).values({ key, value: String(current), updatedAt: new Date().toISOString() })
      .onConflictDoUpdate({ target: kvStore.key, set: { value: String(current), updatedAt: new Date().toISOString() } })
      .run()
    return current
  }
  return stored
}
```
